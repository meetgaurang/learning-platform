CREATE TABLE IF NOT EXISTS users (
	user_id uuid DEFAULT gen_random_uuid() NOT NULL,
	email text NOT NULL,
	firstname text NOT NULL,
	lastname text NULL,
	gender text NULL,
	birthyear int4 NULL,
	city text NULL,
	country text NULL,
	disabled bool NULL,
	CONSTRAINT users_email_key UNIQUE (email),
	CONSTRAINT users_pkey PRIMARY KEY (user_id)
);
